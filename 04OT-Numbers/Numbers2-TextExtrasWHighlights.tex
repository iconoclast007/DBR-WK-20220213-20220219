\marginpar{\scriptsize \centering \fcolorbox{bone}{lime}{\textbf{EACH ONE HIS PLACE}}\\ (Numbers 2:1-34)

\begin{compactenum}[I.][8]
    \item \textbf{Standards} \index[scripture]{Numbers!Num 02:02}\index[scripture]{Numbers!Num 02:03}\index[scripture]{Numbers!Num 02:10}\index[scripture]{Numbers!Num 02:17}\index[scripture]{Numbers!Num 02:18}\index[scripture]{Numbers!Num 01:25}\index[scripture]{Numbers!Num 02:31}\index[scripture]{Numbers!Num 02:34}(Numbers 2:2, 3, 10, 17, 18, 25, 31, 34) 
    \item The \textbf{Sun} \index[scripture]{Numbers!Num 02:03}(Numbers 2:3) -- the Tribe of Judah led the three tribes closest to the Sun (Judah, Issachar, Zebulun)
    \item Four \textbf{Sides} \index[scripture]{Numbers!Num 02:03}\index[scripture]{Numbers!Num 02:10}\index[scripture]{Numbers!Num 02:18}\index[scripture]{Numbers!Num 02:25}(Numbers 2:3,18, 25) 
    \item \textbf{Sons} \index[scripture]{Numbers!Num 02:14}\index[scripture]{Numbers!Num 02:18}\index[scripture]{Numbers!Num 02:22}(Numbers 2:14, 18, 22) 
    \item \textbf{Setting Forth} \index[scripture]{Numbers!Num 02:09}(Numbers 2:9) -- perhaps as a prophetic picture, the tribe of Judah would like Israel when they moved
    \item \textbf{Scene} -- as seen from overhead, the Israelite camp would look like a cross with the Tabernacle in the center. The Most Holy would be in the precise center.
    \item \textbf{Separation} -- Levites were a special case
\end{compactenum}}
